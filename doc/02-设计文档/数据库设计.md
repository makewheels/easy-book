# Easy Book - 泳课学员管理系统数据库设计文档

## 1. 数据库概述

Easy Book系统采用MongoDB作为数据存储解决方案，利用其灵活的文档数据库特性，快速迭代开发，满足轻量级学员管理系统的需求。

### 1.1 技术选型理由

- **MongoDB 6.x+**：NoSQL文档数据库，数据结构灵活
- **PyMongo 4.x+**：MongoDB官方Python驱动，性能优秀
- **文档型存储**：适合学员、预约等半结构化数据
- **水平扩展**：支持未来数据量增长
- **JSON格式**：与前端API天然契合

## 3. 数据关系设计

### 3.1 关系图

```
students (1) ──────── (N) appointments (1) ──────── (N) attendances
   │                      │                           │
   │                      │                           │
   └──────────────────────┘                           │
   │                                                  │
   └──────────────────────────────────────────────────┘
```

### 3.2 关系说明

1. **学员与预约**：一对多关系
   - 一个学员可以有多个预约
   - 一个预约只属于一个学员

2. **预约与上课记录**：一对一关系
   - 一个预约对应一个上课记录
   - 一个上课记录只对应一个预约

3. **学员与上课记录**：一对多关系
   - 一个学员有多个上课记录
   - 一个上课记录只属于一个学员

### 3.3 数据完整性约束

```javascript
// 预约创建时的业务规则
function validateAppointment(appointmentData) {
  // 1. 学员必须存在
  const student = db.students.findOne({ _id: appointmentData.student_id });
  if (!student) {
    throw new Error("学员不存在");
  }
  
  // 2. 1v1课程时间冲突检查
  if (student.package_type === "1v1") {
    const conflict = db.appointments.findOne({
      student_id: { $ne: appointmentData.student_id },
      appointment_date: appointmentData.appointment_date,
      time_slot: appointmentData.time_slot,
      status: { $ne: "cancelled" }
    });
    
    if (conflict) {
      const conflictStudent = db.students.findOne({ _id: conflict.student_id });
      if (conflictStudent.package_type === "1v1") {
        throw new Error("时间段已有其他1v1预约");
      }
    }
  }
  
  // 3. 预约日期不能是过去
  if (appointmentData.appointment_date < new Date()) {
    throw new Error("预约日期不能是过去");
  }
  
  return true;
}

// 签到时的业务规则
function validateCheckin(studentId, appointmentId) {
  // 1. 学员必须有剩余课程
  const student = db.students.findOne({ _id: studentId });
  if (!student || student.remaining_lessons <= 0) {
    throw new Error("剩余课程不足");
  }
  
  // 2. 预约必须存在且未签到
  const appointment = db.appointments.findOne({ _id: appointmentId });
  if (!appointment || appointment.status !== "scheduled") {
    throw new Error("预约状态无效");
  }
  
  // 3. 不能重复签到
  const existing = db.attendances.findOne({
    appointment_id: appointmentId
  });
  if (existing) {
    throw new Error("已经签到过了");
  }
  
  return true;
}
```

```javascript
function createAppointment(appointmentData) {
  // 验证业务规则
  validateAppointment(appointmentData);
  
  // 设置状态和时间
  appointmentData.status = "scheduled";

// 获取每日预约
function getDailyAppointments(date) {
  const startDate = new Date(date);
  const endDate = new Date(date);
  endDate.setDate(endDate.getDate() + 1);
  
  const appointments = db.appointments.find({
    appointment_date: { $gte: startDate, $lt: endDate }
  }).toArray();
  
  // 按时间段分组
  const grouped = {};
  appointments.forEach(apt => {
    if (!grouped[apt.time_slot]) {
      grouped[apt.time_slot] = [];
    }
    
    // 获取学员信息
    const student = db.students.findOne({ _id: apt.student_id });
    grouped[apt.time_slot].push({
      ...apt,
      student_info: {
        name: student.name,
        learning_item: student.learning_item,
        package_type: student.package_type,
        attended_lessons: student.total_lessons - student.remaining_lessons,
        total_lessons: student.total_lessons
      }
    });
  });
  
  return grouped;
}
```

### 4.3 签到操作

```javascript
// 学员签到
function checkIn(studentId, appointmentId) {
  // 验证业务规则
  validateCheckin(studentId, appointmentId);
  
  // 获取学员和预约信息
  const student = db.students.findOne({ _id: studentId });
  const appointment = db.appointments.findOne({ _id: appointmentId });
  
  // 创建上课记录
  const attendanceData = {
    student_id: studentId,
    appointment_id: appointmentId,
    attendance_date: appointment.appointment_date,
    time_slot: appointment.time_slot,
    status: "checked",
    lessons_before: student.remaining_lessons,
    lessons_after: student.remaining_lessons - 1,
    create_time: new Date()
  };
  
  // 开始事务
  const session = db.getMongo().startSession();
  session.startTransaction();
  
  try {
    // 插入上课记录
    db.attendances.insertOne(attendanceData, { session });
    
    // 更新学员剩余课程
    db.students.updateOne(
      { _id: studentId },
      { 
        $inc: { remaining_lessons: -1 },
        $set: { update_time: new Date() }
      },
      { session }
    );
    
    // 更新预约状态
    db.appointments.updateOne(
      { _id: appointmentId },
      { 
        $set: { 
          status: "checked",
          update_time: new Date()
        }
      },
      { session }
    );
    
    // 提交事务
    session.commitTransaction();
    
    return attendanceData;
  } catch (error) {
    // 回滚事务
    session.abortTransaction();
    throw error;
  } finally {
    session.endSession();
  }
}
```

```javascript
// 优化前：获取学员预约记录（多次查询）
function getAppointmentsByStudentSlow(studentId) {
  const appointments = db.appointments.find({ student_id: studentId }).toArray();
  
  return appointments.map(apt => {
    const student = db.students.findOne({ _id: apt.student_id });
    return { ...apt, student_info: student };
  });
}

// 优化后：使用聚合管道（一次查询）
function getAppointmentsByStudentFast(studentId) {
  return db.appointments.aggregate([
    { $match: { student_id: ObjectId(studentId) } },
    {
      $lookup: {
        from: "students",
        localField: "student_id",
        foreignField: "_id",
        as: "student_info"
      }
    },
    { $unwind: "$student_info" },
    {
      $project: {
        student_info: {
          name: 1,
          learning_item: 1,
          package_type: 1,
          attended_lessons: { $subtract: ["$student_info.total_lessons", "$student_info.remaining_lessons"] },
          total_lessons: "$student_info.total_lessons"
        },
        appointment_date: 1,
        time_slot: 1,
        status: 1
      }
    }
  ]).toArray();
}
```

```javascript
// 高效分页查询
function getStudentsWithPagination(page, size, filters = {}) {
  const skip = (page - 1) * size;
  
  // 构建查询条件
  const query = {};
  if (filters.keyword) {
    query.$or = [
      { name: { $regex: filters.keyword, $options: "i" } },
      { nickname: { $regex: filters.keyword, $options: "i" } }
    ];
  }
  if (filters.learning_item) {
    query.learning_item = filters.learning_item;
  }
  if (filters.package_type) {
    query.package_type = filters.package_type;
  }
  
  // 执行查询
  const students = db.students
    .find(query)
    .sort({ create_time: -1 })
    .skip(skip)
    .limit(size)
    .toArray();
  
  // 获取总数
  const total = db.students.countDocuments(query);
  
  return {
    items: students,
    total,
    page,
    size,
    pages: Math.ceil(total / size)
  };
}
```

```javascript
class CachedDataService {
  async getStudentWithCache(studentId) {
    const cacheKey = `student:${studentId}`;
    
    // 尝试从缓存获取
    let student = await redis.get(cacheKey);
    if (student) {
      return JSON.parse(student);
    }
    
    // 从数据库获取
    student = db.students.findOne({ _id: ObjectId(studentId) });
    
    // 写入缓存（5分钟过期）

      { _id: ObjectId(studentId) },
      { $set: { ...updateData, update_time: new Date() } }
    );
    
    // 清除缓存
    await redis.del(`student:${studentId}`);
    
    return result;
  }
```

```javascript
    appointment_date: { $lt: oneYearAgo },
    status: { $in: ["checked", "absent"] }
  });
  
  console.log(`清理了 ${appointmentResult.deletedCount} 条预约记录`);
}
```

### 7.3 数据一致性检查

```javascript
// 检查数据一致性
function checkDataConsistency() {
  const issues = [];
  
  // 检查孤立的预约记录
  const orphanAppointments = db.appointments.aggregate([
    {
      $lookup: {
        from: "students",
        localField: "student_id",
        foreignField: "_id",
        as: "student"
      }
    },
    { $match: { student: { $size: 0 } } }
  ]).toArray();
  
  if (orphanAppointments.length > 0) {
    issues.push(`发现 ${orphanAppointments.length} 条孤立预约记录`);
  }
  
  // 检查孤立的上课记录
  const orphanAttendances = db.attendances.aggregate([
    {
      $lookup: {
        from: "students",
        localField: "student_id",
        foreignField: "_id",
        as: "student"
      }
    },
    { $match: { student: { $size: 0 } } }
  ]).toArray();
  
  if (orphanAttendances.length > 0) {
    issues.push(`发现 ${orphanAttendances.length} 条孤立上课记录`);
  }
  
  // 检查剩余课程数一致性
  const students = db.students.find({}).toArray();
  students.forEach(student => {
    const attendances = db.attendances.find({
      student_id: student._id,
      status: "checked"
    }).toArray();
    
    const expectedRemaining = student.total_lessons - attendances.length;
    if (student.remaining_lessons !== expectedRemaining) {
      issues.push(`学员 ${student.name} 剩余课程数不一致：期望 ${expectedRemaining}，实际 ${student.remaining_lessons}`);
    }
  });
  
  return issues;
}
```
